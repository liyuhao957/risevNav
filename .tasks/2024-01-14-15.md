## Shell脚本实现细节
1. 工具函数 (utils.sh)
   ```bash
   # 颜色定义
   RED='\033[0;31m'
   GREEN='\033[0;32m'
   YELLOW='\033[1;33m'
   NC='\033[0m'

   # 检查端口是否被占用
   check_port() {
     lsof -i:$1 > /dev/null 2>&1
     return $?
   }

   # 检查服务是否启动成功
   check_service() {
     curl -s http://localhost:$1/health > /dev/null
     return $?
   }

   # 日志函数
   log() {
     echo -e "${2:-$GREEN}[$(date '+%Y-%m-%d %H:%M:%S')] $1${NC}" | tee -a "$LOG_FILE"
   }
   ```

     # 检查 Node.js

       log "Node.js is required but not installed." "$RED"
       exit 1
     }
     # 检查 Python
     command -v python3 >/dev/null 2>&1 || {
       log "Python 3 is required but not installed." "$RED"
       exit 1
     }
   }

   # 2. 启动后端服务
   start_backend() {
     cd "$NAV_BACKEND_DIR" && npm run dev &
     cd "$MON_BACKEND_DIR" && npm run dev &
   }

   # 3. 启动前端服务
   start_frontend() {
     cd "$NAV_FRONTEND_DIR" && npm run serve &
     cd "$MON_FRONTEND_DIR" && npm run dev &
   }

   # 4. 启动监控脚本
